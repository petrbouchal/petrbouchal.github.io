source("library.R")
source("loadprocesseddata.r")
library(ggplot2)
library(ggiraph)
kurzviolinplot <- kurzy %>% group_by(session, ktery) %>%
  filter(!is.na(hodnoceni)) %>% 
  summarise(prumer = mean(as.numeric(hodnoceni), na.rm=T),
            stdev = sd(hodnoceni, na.rm=T),
            pocet = n()) %>%
  mutate(dataid = row_number()) %>%
  arrange(desc(pocet)) %>% 
  ggplot(aes(session, prumer, alpha=.2)) +
  geom_violin(fill="grey", colour=NA) +
  geom_point_interactive(aes(size=pocet, fill=stdev, colour=stdev,
                 tooltip=ktery, data_id=dataid), shape=21) +
  scale_fill_continuous(low = "yellow",high = "red") +
  scale_colour_continuous(low = "yellow",high = "red")  + theme_discover +
  scale_alpha(guide="none") +
  theme(legend.position = "bottom", legend.direction = "horizontal",
        legend.box="horizontal", legend.spacing = unit(3,"mm"))
## Warning: Ignoring unknown aesthetics: tooltip, data_id
ggiraph(print(kurzviolinplot), tooltip_extra_css=tooltip_css, hover_css = "opacity: .5;")
